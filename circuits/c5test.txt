input Int64 a, b, c

output Int32 z, x

wire Int64 d, e, f, g, h 
wire Int1 dLTe, dEQe 
wire Int64 xrin, zrin
register Int64 greg, hreg

// a = 15; b = 27; c = 33; => z =42, x =-24;
d = a + b //15 + 27 = 42
e = a + c //15 + 33 = 48
f = a - b   //15 - 27 = -12
dEQe = d == e //false = 0
dLTe = d < e // true = 1
g = dLTe ? d : e //g = d = 42  //If true: if false
h = dEQe ? g : f //h = g = -12
greg = g //42
hreg = h //-12
xrin = hreg << dLTe //48 = 
zrin = greg >> dEQe
x = xrin
z = zrin
